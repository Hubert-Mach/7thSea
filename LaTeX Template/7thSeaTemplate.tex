% Define custom keys for tables
\keys_define:nn { custom / table } {
	color .tl_set:N = \l__custom_table_color_tl,
	color .initial:n = white,
	color .value_required:n = true,
	header .tl_set:N = \l__custom_table_header_tl,
	header .value_required:n = true,
	width .dim_set:N = \l__custom_table_width_dim,
	width .value_required:n = true,
}

% Nice Table environment
\NewDocumentEnvironment {NiceTable} { o m } {
	\group_begin:
	% Set default width to linewidth
	\dim_set:Nn \l__custom_table_width_dim { \linewidth }
	
	% Process optional arguments
	\tl_if_novalue:nF {#1} { \keys_set:nn { custom / table } {#1} }
	
	% Add spacing before table
	\par \vspace { 9pt plus 3pt minus 3pt } \noindent
	
	% Process header if provided
	\tl_if_empty:NF \l__custom_table_header_tl {
		\group_begin:
		\large\bfseries\sffamily \l__custom_table_header_tl \nopagebreak
		\par \vspace{ 5pt plus 2pt minus 2pt } \noindent
		\group_end:
	}
	
	% Apply font settings
	\small % Set font size to small
	\sffamily % Switch to sans-serif font family
	
	% Set up row colors
	\rowcolors {1} {\l__custom_table_color_tl} {}
	
	% Begin tabularx environment
	\tabularx {\l__custom_table_width_dim} {#2}
} {
	\endtabularx 
	\vspace { 9pt plus 3pt minus 3pt }
	\group_end:
}
